var name = 'Manish'
function multiply(a,b){
    document.write('Hey ' + this.name + ', Mulplication of ' + a + ' and ' + b + ' is ' +  a*b + '<br/>');
}

var ema = {
    name : 'Ema'
}

multiplyBy3(3)
